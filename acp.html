def min_note(amount:float, notes_list:list, notes_type:str):
    print(notes_type.upper()+"S:")
    for note in notes_list:
        print(f"{int(amount/note)} {notes_type}(s) de R$ {note:.2f}")
        amount = round(amount%note, 2)
    return amount

value = float(input())
notes = [100, 50, 20, 10, 5, 2]
coins = [1, 0.50, 0.25, 0.10, 0.05, 0.01]
value = min_note(value, notes, "nota")
min_note(value, coins, "moeda")
